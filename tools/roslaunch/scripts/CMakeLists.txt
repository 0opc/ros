file(MAKE_DIRECTORY ${CMAKE_BINARY_DIR}/bin)
if(NOT EXISTS ${CMAKE_BINARY_DIR}/bin/roslaunch)
  execute_process(COMMAND /bin/ln -s ${CMAKE_CURRENT_SOURCE_DIR}/roslaunch ${CMAKE_BINARY_DIR}/bin/roslaunch)
endif()
if(NOT EXISTS ${CMAKE_BINARY_DIR}/bin/roslaunch-deps)
  execute_process(COMMAND /bin/ln -s ${CMAKE_CURRENT_SOURCE_DIR}/roslaunch-deps ${CMAKE_BINARY_DIR}/bin/roslaunch-deps)
endif()
if(NOT EXISTS ${CMAKE_BINARY_DIR}/bin/roslaunch-logs)
  execute_process(COMMAND /bin/ln -s ${CMAKE_CURRENT_SOURCE_DIR}/roslaunch-logs ${CMAKE_BINARY_DIR}/bin/roslaunch-logs)
endif()
if(NOT EXISTS ${CMAKE_BINARY_DIR}/bin/roscore)
  execute_process(COMMAND /bin/ln -s ${CMAKE_CURRENT_SOURCE_DIR}/roscore ${CMAKE_BINARY_DIR}/bin/roscore)
endif()
